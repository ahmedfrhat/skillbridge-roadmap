// كود التمرير والتعرف التلقائي بـ http.ts

if (typeof window !== "undefined") {
  const token = localStorage.getItem("skillbridge_token");
  if (token && !headers.has("Authorization")) {
    headers.set("Authorization", `Bearer ${token}`);
  }
}

AuthMiddleware.ts كود معتمد 
export const authenticateJWT = (req: Request, res: Response, next: NextFunction) => {
  const authHeader = req.headers.authorization;
  if (authHeader) {
    const token = authHeader.split(' ')[1];
    jwt.verify(token, JWT_SECRET, (err, user) => {
      if (err) return res.sendStatus(403);
      req.user = user;
      next();
    });
  } else {
    res.sendStatus(401);
  }
};

1. نظام اعتماد التدريب الميداني الجامعي

تقوم الجامعات المصرية بطلب تدريب صيفي معتمد للطلاب كقيد للتخرج. قمنا بدمج دورة عمل مخصصة لأساتذة الجامعات (Academic Supervisors)؛ حيث يرسل الطالب مشروعه المنجز للأستاذ لمراجعته، ليقوم الأستاذ باعتماده وتوثيقه كـ ساعات تدريب صيفي معتمدة برقم تسلسلي موحد!

لحماية متبادلة للحقوق، قمنا بإعداد ميزة تقوم تلقائياً بصياغة عقد عمل قانوني مبسط باللغة العربية (SLA) يغطي: بنود الملكية الفكرية للكود، مواعيد التسليم والنسب، وشروط NDA الحساسة، بالاعتماد على ميزانية المشروع وحجم الفريق، ليوثق العقد بضغطة موافقة من الطرفين قبل الدفع.

3. محاكي الأكواد الحي (Sandbox Code Runner)

لتأكيد المهارة العملية وتخفيف الشك، نوفر مستعرضاً حياً للأكواد في ملفات الطلاب؛ حيث تتيح الواجهة للعميل تشغيل واختبار ملفات الكود التي رفعها الطلاب للتطبيق في بيئة افتراضية تفاعلية حية (Sandbox Container).
